1 项目背景

本分析报告旨在复现 Kimura et al. (Nature 2025) 中关于小鼠皮层(Cortex)单细胞转录组数据分析的主要步骤。 原始数据来源于 GEO 数据库：GSE267764。 通过标准的 Seurat v5 流程，我们对数据进行了质量控制、降维聚类、细胞亚群注释、双细胞去除以及差异表达分析 (DGE)。 —

  y <- fs[grepl(x, fs)]

})

4 降维聚类分析与批次效应校正

在此模块中，我们将数据标准化，寻找高变基因，并利用 PCA 进行线性降维。由于存在多个样本，我们使用 harmony 算法对样本间可能存在的批次效应进行校正，随后进行非线性降维 (UMAP) 和图聚类 (Louvain)。

Code 
library(harmony)

# 封装一个标准的降维聚类 Pipeline 函数
scrna_process <- function(seurat_obj, npc, res, normalize = TRUE, run_harmony = TRUE) {
  # 1. 数据标准化：消除测序深度差异
  if (normalize) {
    seurat_obj <- NormalizeData(seurat_obj)
  }
  
  # 2. 寻找高变基因 (Top 2000)，用于捕捉细胞间主要异质性
  seurat_obj <- FindVariableFeatures(seurat_obj, selection.method = "vst", nfeatures = 2000)
  
  # 3. 数据缩放并执行主成分分析 (PCA)
  seurat_obj <- ScaleData(seurat_obj, features = VariableFeatures(seurat_obj)) %>%
    RunPCA(features = VariableFeatures(seurat_obj))
  
  # 4. 根据是否需要批次校正执行不同分支
  if (run_harmony) {
    # 使用 Harmony 根据样本来源 (orig.ident) 整合批次
    seurat_obj <- RunHarmony(seurat_obj, group.by.vars = "orig.ident")
    # 基于整合后的 harmony 空间进行 UMAP 和 KNN 构建
    seurat_obj <- RunUMAP(seurat_obj, reduction = "harmony", dims = 1:npc)
    seurat_obj <- FindNeighbors(seurat_obj, reduction = "harmony", dims = 1:npc)
    seurat_obj <- FindClusters(seurat_obj, resolution = res)
  } else {
    # 不校正批次效应的标准流程
    seurat_obj <- seurat_obj %>%
      FindNeighbors(dims = 1:npc) %>%
      FindClusters(resolution = res) %>%
      RunUMAP(dims = 1:npc)
  }
  return(seurat_obj)
}

# 运行自定义的 Pipeline，选取前 50 个 PC，聚类分辨率设定为 0.4
scRNAseq_cortex <- scrna_process(scRNAseq_cortex, npc = 50, res = 0.4)

# 绘制初次聚类的 UMAP 图
DimPlot(scRNAseq_cortex, label = TRUE)

6 双细胞检测与过滤 (DoubletFinder)

单细胞测序中一个液滴包裹多个细胞（Doublets）会导致假性细胞类的产生。这里我们使用经典的 DoubletFinder 算法进行鉴定。 注：为了演示效率，代码中对数据进行了 10,000 个细胞的抽样。针对 Seurat v5，代码非常规范地使用了 LayerData 来兼容图层机制。

Code 
library(DoubletFinder)

# 完善实验设计分组元数据 (Control vs KO)
if(!"condition" %in% colnames(scRNAseq_cortex@meta.data)){
  scRNAseq_cortex$condition <- ifelse(grepl("KO", scRNAseq_cortex$orig.ident), "cko", "control")
}

# 为了节省计算资源进行演示，随机抽取 10000 个细胞
set.seed(123)
n_cells <- 10000
sampled_cells <- sample(colnames(scRNAseq_cortex), size = n_cells)
sc_sub_cortex <- scRNAseq_cortex[, sampled_cells]

# ========== 关键修正：解决 Seurat v5 图层兼容性问题 ==========
# 在 Seurat v5 中，数据往往分布在多个 layer 中（例如按样本分层的 counts）
# 提取合并后的 counts 层数据可以避免 DoubletFinder 报错
counts_sub <- LayerData(sc_sub_cortex, assay = "RNA", layer = "counts")
metadata_sub <- sc_sub_cortex@meta.data

# 新建一个单层的对象进行双细胞检测
sc_sub_simple <- CreateSeuratObject(counts = counts_sub, meta.data = metadata_sub, project = "Doublet_sub")

# 对子集执行基础处理以提供给 DoubletFinder (必须具有 PCA 降维结果)
sc_sub_simple <- NormalizeData(sc_sub_simple)
sc_sub_simple <- FindVariableFeatures(sc_sub_simple, selection.method = "vst", nfeatures = 2000)
sc_sub_simple <- ScaleData(sc_sub_simple, features = VariableFeatures(sc_sub_simple))
sc_sub_simple <- RunPCA(sc_sub_simple, features = VariableFeatures(sc_sub_simple))
sc_sub_simple <- FindNeighbors(sc_sub_simple, dims = 1:50)
sc_sub_simple <- FindClusters(sc_sub_simple, resolution = 0.4)
sc_sub_simple <- RunUMAP(sc_sub_simple, dims = 1:50) 

# ========== 运行 DoubletFinder ==========
# 1. 扫描 pK 参数：寻找人工构建双细胞的最优邻域大小
sweep.res <- paramSweep(sc_sub_simple, PCs = 1:50, sct = FALSE)
sweep.stats <- summarizeSweep(sweep.res, GT = FALSE)
bcmvn <- find.pK(sweep.stats)
# 选取 BCmetric 最大的 pK 值
pK <- as.numeric(as.character(bcmvn$pK[which.max(bcmvn$BCmetric)]))

# 2. 估算同源双细胞比例（Homotypic doublets，即同种细胞类型的融合，算法较难检测）
clusters <- Idents(sc_sub_simple) 
homotypic.prop <- modelHomotypic(clusters)

# 3. 计算预期需要剔除的双细胞数目（假设 10X 平台的双细胞率约 7.5%）
nExp_poi <- round(0.075 * nrow(sc_sub_simple@meta.data))
nExp_poi.adj <- round(nExp_poi * (1 - homotypic.prop))

# 4. 执行核心鉴定算法
sc_sub_simple <- doubletFinder(sc_sub_simple, PCs = 1:50, pK = pK, nExp = nExp_poi.adj)

# ========== 提取结果并映射回原始数据 ==========
# 动态获取 DoubletFinder 生成的结果列名
df_colname <- grep("DF.classifications", colnames(sc_sub_simple@meta.data), value = TRUE)[1]
sc_sub_cortex$doublet_status <- sc_sub_simple@meta.data[[df_colname]]

# 结果可视化
DimPlot(sc_sub_simple, group.by = df_colname, reduction = "umap", label = TRUE) + 
  ggtitle("Doublet detection (subset)")

7 差异表达分析 (DGE)

我们希望找出在 KO (cko) 组和对照组 (control) 之间表达有显著差异的基因。这里对所有定义的细胞亚群进行了循环检验。同样地，针对 Seurat v5 采用了 JoinLayers 的优秀实践。

Code 
library(tibble)

# 检查分组信息
if(!"condition" %in% colnames(scRNAseq_cortex@meta.data)){
  scRNAseq_cortex$condition <- ifelse(grepl("KO", scRNAseq_cortex$orig.ident), "cko", "control")
}

# 定义针对多细胞类型的差异分析函数
scrna_DGE <- function(SeuratData, logfc.threshold = 0.25, min.pct = 0.3, test.use = "wilcox") {
  cell_type <- levels(as.factor(SeuratData$cell_type))
  
  # 对每种细胞类型循环执行 Wilcoxon 秩和检验
  results <- lapply(cell_type, function(x) {
    # 提取当前细胞类型
    subset_obj <- subset(SeuratData, cell_type == x)
    
    # 关键修复：Seurat v5 需要先合并拆分的图层才能进行差异分析
    subset_obj <- JoinLayers(subset_obj)
    
    # 在当前细胞类型内，比较 cko 组和 control 组
    res <- FindMarkers(subset_obj, 
                       ident.1 = "cko", ident.2 = "control", group.by = "condition",
                       logfc.threshold = logfc.threshold, min.pct = min.pct, test.use = test.use)
    
    if (nrow(res) == 0) return(NULL)
    
    # 结果格式化与统计量整理
    res %>% 
      rownames_to_column("gene") %>%
      mutate(direction = ifelse(avg_log2FC > 0, "up", "down")) %>%
      mutate(direction = factor(direction, c("up", "down"))) %>%
      mutate(FDR = p.adjust(p_val, method = "BH")) %>% 
      mutate(Celltype = rep(x, nrow(res)))
  })
  
  names(results) <- cell_type
  return(results)
}

# 运行差异分析
SCRNAseq_DGE <- scrna_DGE(scRNAseq_cortex)
combined_DGE <- do.call(rbind, SCRNAseq_DGE)

# 筛选显著差异基因：校正P值 < 0.05, 对数差异倍数 > 0.25, 表达比例 > 30%
combined_DGE_filter <- combined_DGE %>% 
  filter(p_val_adj < 0.05 & abs(avg_log2FC) > 0.25 & pct.1 > 0.3)

library(ggrepel)

  mutate(Difference = pct.1 - pct.2) %>% # 计算两组表达比例的差值
  rownames_to_column("gene")

#  降维聚类分析与批次效应校正

在此模块中，我们将数据标准化，寻找高变基因，并利用 PCA 进行线性降维。由于存在多个样本，我们使用 `harmony` 算法对样本间可能存在的批次效应进行校正，随后进行非线性降维 (UMAP) 和图聚类 (Louvain)。

```{r}
#| label: dimensionality-reduction

library(harmony)

# 封装一个标准的降维聚类 Pipeline 函数
scrna_process <- function(seurat_obj, npc, res, normalize = TRUE, run_harmony = TRUE) {
  # 1. 数据标准化：消除测序深度差异
  if (normalize) {
    seurat_obj <- NormalizeData(seurat_obj)
  }
  
  # 2. 寻找高变基因 (Top 2000)，用于捕捉细胞间主要异质性
  seurat_obj <- FindVariableFeatures(seurat_obj, selection.method = "vst", nfeatures = 2000)
  
  # 3. 数据缩放并执行主成分分析 (PCA)
  seurat_obj <- ScaleData(seurat_obj, features = VariableFeatures(seurat_obj)) %>%
    RunPCA(features = VariableFeatures(seurat_obj))
  
  # 4. 根据是否需要批次校正执行不同分支
  if (run_harmony) {
    # 使用 Harmony 根据样本来源 (orig.ident) 整合批次
    seurat_obj <- RunHarmony(seurat_obj, group.by.vars = "orig.ident")
    # 基于整合后的 harmony 空间进行 UMAP 和 KNN 构建
    seurat_obj <- RunUMAP(seurat_obj, reduction = "harmony", dims = 1:npc)
    seurat_obj <- FindNeighbors(seurat_obj, reduction = "harmony", dims = 1:npc)
    seurat_obj <- FindClusters(seurat_obj, resolution = res)
  } else {
    # 不校正批次效应的标准流程
    seurat_obj <- seurat_obj %>%
      FindNeighbors(dims = 1:npc) %>%
      FindClusters(resolution = res) %>%
      RunUMAP(dims = 1:npc)
  }
  return(seurat_obj)
}

# 运行自定义的 Pipeline，选取前 50 个 PC，聚类分辨率设定为 0.4
scRNAseq_cortex <- scrna_process(scRNAseq_cortex, npc = 50, res = 0.4)

# 绘制初次聚类的 UMAP 图
DimPlot(scRNAseq_cortex, label = TRUE)

# 剔除在 UMAP 空间上呈现离群状态的低质量亚群 (Cluster 6)
scRNAseq_cortex <- subset(scRNAseq_cortex, seurat_clusters %in% c(0:5, 7:10))
DimPlot(scRNAseq_cortex, label = TRUE)
```

#  双细胞检测与过滤 (DoubletFinder)

单细胞测序中一个液滴包裹多个细胞（Doublets）会导致假性细胞类的产生。这里我们使用经典的 `DoubletFinder` 算法进行鉴定。 *注：为了演示效率，代码中对数据进行了 10,000 个细胞的抽样。针对 Seurat v5，代码非常规范地使用了 `LayerData` 来兼容图层机制。*

```{r}
#| label: doublet-detection
#| fig-width: 8
#| fig-height: 4

library(DoubletFinder)

# 完善实验设计分组元数据 (Control vs KO)
if(!"condition" %in% colnames(scRNAseq_cortex@meta.data)){
  scRNAseq_cortex$condition <- ifelse(grepl("KO", scRNAseq_cortex$orig.ident), "cko", "control")
}

# 为了节省计算资源进行演示，随机抽取 10000 个细胞
set.seed(123)
n_cells <- 10000
sampled_cells <- sample(colnames(scRNAseq_cortex), size = n_cells)
sc_sub_cortex <- scRNAseq_cortex[, sampled_cells]

# ========== 关键修正：解决 Seurat v5 图层兼容性问题 ==========
# 在 Seurat v5 中，数据往往分布在多个 layer 中（例如按样本分层的 counts）
# 提取合并后的 counts 层数据可以避免 DoubletFinder 报错
counts_sub <- LayerData(sc_sub_cortex, assay = "RNA", layer = "counts")
metadata_sub <- sc_sub_cortex@meta.data

# 新建一个单层的对象进行双细胞检测
sc_sub_simple <- CreateSeuratObject(counts = counts_sub, meta.data = metadata_sub, project = "Doublet_sub")

# 对子集执行基础处理以提供给 DoubletFinder (必须具有 PCA 降维结果)
sc_sub_simple <- NormalizeData(sc_sub_simple)
sc_sub_simple <- FindVariableFeatures(sc_sub_simple, selection.method = "vst", nfeatures = 2000)
sc_sub_simple <- ScaleData(sc_sub_simple, features = VariableFeatures(sc_sub_simple))
sc_sub_simple <- RunPCA(sc_sub_simple, features = VariableFeatures(sc_sub_simple))
sc_sub_simple <- FindNeighbors(sc_sub_simple, dims = 1:50)
sc_sub_simple <- FindClusters(sc_sub_simple, resolution = 0.4)
sc_sub_simple <- RunUMAP(sc_sub_simple, dims = 1:50) 

# ========== 运行 DoubletFinder ==========
# 1. 扫描 pK 参数：寻找人工构建双细胞的最优邻域大小
sweep.res <- paramSweep(sc_sub_simple, PCs = 1:50, sct = FALSE)
sweep.stats <- summarizeSweep(sweep.res, GT = FALSE)
bcmvn <- find.pK(sweep.stats)
# 选取 BCmetric 最大的 pK 值
pK <- as.numeric(as.character(bcmvn$pK[which.max(bcmvn$BCmetric)]))

# 2. 估算同源双细胞比例（Homotypic doublets，即同种细胞类型的融合，算法较难检测）
clusters <- Idents(sc_sub_simple) 
homotypic.prop <- modelHomotypic(clusters)

# 3. 计算预期需要剔除的双细胞数目（假设 10X 平台的双细胞率约 7.5%）
nExp_poi <- round(0.075 * nrow(sc_sub_simple@meta.data))
nExp_poi.adj <- round(nExp_poi * (1 - homotypic.prop))

# 4. 执行核心鉴定算法
sc_sub_simple <- doubletFinder(sc_sub_simple, PCs = 1:50, pK = pK, nExp = nExp_poi.adj)

# ========== 提取结果并映射回原始数据 ==========
# 动态获取 DoubletFinder 生成的结果列名
df_colname <- grep("DF.classifications", colnames(sc_sub_simple@meta.data), value = TRUE)[1]
sc_sub_cortex$doublet_status <- sc_sub_simple@meta.data[[df_colname]]

# 结果可视化
DimPlot(sc_sub_simple, group.by = df_colname, reduction = "umap", label = TRUE) + 
  ggtitle("Doublet detection (subset)")
```

#  差异表达分析 (DGE)

我们希望找出在 KO (cko) 组和对照组 (control) 之间表达有显著差异的基因。这里对所有定义的细胞亚群进行了循环检验。同样地，针对 Seurat v5 采用了 `JoinLayers` 的优秀实践。

```{r}
#| label: differential-expression

library(tibble)

# 检查分组信息
if(!"condition" %in% colnames(scRNAseq_cortex@meta.data)){
  scRNAseq_cortex$condition <- ifelse(grepl("KO", scRNAseq_cortex$orig.ident), "cko", "control")
}

# 定义针对多细胞类型的差异分析函数
scrna_DGE <- function(SeuratData, logfc.threshold = 0.25, min.pct = 0.3, test.use = "wilcox") {
  cell_type <- levels(as.factor(SeuratData$cell_type))
  
  # 对每种细胞类型循环执行 Wilcoxon 秩和检验
  results <- lapply(cell_type, function(x) {
    # 提取当前细胞类型
    subset_obj <- subset(SeuratData, cell_type == x)
    
    # 关键修复：Seurat v5 需要先合并拆分的图层才能进行差异分析
    subset_obj <- JoinLayers(subset_obj)
    
    # 在当前细胞类型内，比较 cko 组和 control 组
    res <- FindMarkers(subset_obj, 
                       ident.1 = "cko", ident.2 = "control", group.by = "condition",
                       logfc.threshold = logfc.threshold, min.pct = min.pct, test.use = test.use)
    
    if (nrow(res) == 0) return(NULL)
    
    # 结果格式化与统计量整理
    res %>% 
      rownames_to_column("gene") %>%
      mutate(direction = ifelse(avg_log2FC > 0, "up", "down")) %>%
      mutate(direction = factor(direction, c("up", "down"))) %>%
      mutate(FDR = p.adjust(p_val, method = "BH")) %>% 
      mutate(Celltype = rep(x, nrow(res)))
  })
  
  names(results) <- cell_type
  return(results)
}

# 运行差异分析
SCRNAseq_DGE <- scrna_DGE(scRNAseq_cortex)
combined_DGE <- do.call(rbind, SCRNAseq_DGE)

# 筛选显著差异基因：校正P值 < 0.05, 对数差异倍数 > 0.25, 表达比例 > 30%
combined_DGE_filter <- combined_DGE %>% 
  filter(p_val_adj < 0.05 & abs(avg_log2FC) > 0.25 & pct.1 > 0.3)
```
